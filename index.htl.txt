    <script>
        // ⚠️ ATENÇÃO: Mude para o SEU número abaixo (DDI + DDD + Número) sem espaços ou traços.
        // Exemplo: "5581999999999" (55 do Brasil, código de área e o número)
        const SEU_NUMERO = "5581992304256"; 

        // Esta frase com o "Te amo ❤️" garante a animação de corações no WhatsApp
        const textoMensagem = "Eu aceito! Feliz Dia dos Namorados, te amo! ❤️🌹";
        const MENSAGEM_FORMATADA = encodeURIComponent(textoMensagem);

        function fuga() {
            const botaoNao = document.getElementById('btn-nao');
            
            // Calcula limites seguros para o botão não sumir completamente da tela
            const larguraJanela = window.innerWidth;
            const alturaJanela = window.innerHeight;
            
            const novoX = Math.random() * (larguraJanela - botaoNao.offsetWidth - 40) + 20;
            const novoY = Math.random() * (alturaJanela - botaoNao.offsetHeight - 40) + 20;
            
            botaoNao.style.position = 'fixed';
            botaoNao.style.left = novoX + 'px';
            botaoNao.style.top = novoY + 'px';
        }

        function aceitou() {
            // Abre o WhatsApp direto no seu chat com a mensagem preenchida
            window.location.href = `https://api.whatsapp.com/send?phone=${SEU_NUMERO}&text=${MENSAGEM_FORMATADA}`;
        }
    </script>
